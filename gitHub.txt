--WHEN PUSHING THE CHANGES YOU MAKE--

git checkout -b 'Documentation_1.0'
git add .
git commit -m "Docu"
git push
git push --set-upstream origin [name_of_the_branch]

--IF MINIMAL PUSHING--
git add .
git commit -m "platform-ginawa_mo"
git push

--FOR UPDATING THE FILE IN THE FOLDER--

git pull origin master

Note: master is the main branch, if you want to clone your latest changes just change the master into your branch


--CLONING THE PROJECT--
git clone -b master https://github.com/jephlandicho/NutriWISE.git

Note: master is the main branch, if you want to clone your latest changes just change the master into your branch
